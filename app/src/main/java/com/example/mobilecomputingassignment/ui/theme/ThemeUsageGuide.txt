// COLORS - Use MaterialTheme.colorScheme for automatic light/dark adaptation

@Composable
fun ColorExamples() {
    Column {
        // Primary Actions (Orange)
        Button(
            onClick = {},
            colors = ButtonDefaults.buttonColors(
                containerColor = MaterialTheme.colorScheme.primary  // Orange
            )
        ) {
            Text("Watch Movie")  // White text automatically
        }

        // Secondary Actions (Yellow)
        Button(
            onClick = {},
            colors = ButtonDefaults.buttonColors(
                containerColor = MaterialTheme.colorScheme.secondary  // Yellow
            )
        ) {
            Text(
                "Add to Watchlist",
                color = MaterialTheme.colorScheme.onSecondary  // Black text on yellow
            )
        }

        // Surface Colors (Cards, Dialogs)
        Card(
            colors = CardDefaults.cardColors(
                containerColor = MaterialTheme.colorScheme.surface  // Adapts to theme
            )
        ) {
            Text(
                "Movie Details",
                color = MaterialTheme.colorScheme.onSurface  // Contrasting text
            )
        }

        // Background Text
        Text(
            "Welcome to WatchMates",
            color = MaterialTheme.colorScheme.onBackground  // Adapts to theme
        )

        // Error States
        Text(
            "Connection failed",
            color = MaterialTheme.colorScheme.error  // Red
        )
    }
}

// COMPLETE COMPONENT EXAMPLES

@Composable
fun MovieCard(
    title: String,
    description: String,
    rating: Float,
    onWatchClick: () -> Unit,
    onAddToListClick: () -> Unit
) {
    Card(
        modifier = Modifier
            .fillMaxWidth()
            .padding(16.dp),
        colors = CardDefaults.cardColors(
            containerColor = MaterialTheme.colorScheme.surface
        ),
        elevation = CardDefaults.cardElevation(defaultElevation = 4.dp)
    ) {
        Column(
            modifier = Modifier.padding(16.dp)
        ) {
            // Movie Title - Racing Sans One
            Text(
                text = title,
                style = MaterialTheme.typography.titleLarge,
                color = MaterialTheme.colorScheme.onSurface
            )

            Spacer(modifier = Modifier.height(8.dp))

            // Rating Row
            Row(
                verticalAlignment = Alignment.CenterVertically
            ) {
                Text(
                    text = "â˜…".repeat(rating.toInt()),
                    style = MaterialTheme.typography.bodyMedium,
                    color = MaterialTheme.colorScheme.secondary  // Yellow stars
                )
                Text(
                    text = " $rating/5",
                    style = MaterialTheme.typography.bodyMedium,
                    color = MaterialTheme.colorScheme.onSurface
                )
            }

            Spacer(modifier = Modifier.height(12.dp))

            // Description - Noto Sans
            Text(
                text = description,
                style = MaterialTheme.typography.bodyMedium,
                color = MaterialTheme.colorScheme.onSurface,
                maxLines = 3
            )

            Spacer(modifier = Modifier.height(16.dp))

            // Action Buttons
            Row(
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                Button(
                    onClick = onWatchClick,
                    colors = ButtonDefaults.buttonColors(
                        containerColor = MaterialTheme.colorScheme.primary  // Orange
                    )
                ) {
                    Text(
                        "Watch Now",
                        style = MaterialTheme.typography.labelLarge
                    )
                }

                OutlinedButton(
                    onClick = onAddToListClick,
                    colors = OutlinedButtonDefaults.buttonColors(
                        contentColor = MaterialTheme.colorScheme.secondary  // Yellow text
                    ),
                    border = BorderStroke(
                        1.dp,
                        MaterialTheme.colorScheme.secondary  // Yellow border
                    )
                ) {
                    Text(
                        "Add to List",
                        style = MaterialTheme.typography.labelLarge
                    )
                }
            }
        }
    }
}

// QUICK REFERENCE
/*
COLORS:
- Primary actions: MaterialTheme.colorScheme.primary (Orange)
- Secondary actions: MaterialTheme.colorScheme.secondary (Yellow)
- Error states: MaterialTheme.colorScheme.error (Red)
- Backgrounds: MaterialTheme.colorScheme.background
- Surfaces: MaterialTheme.colorScheme.surface
- Text: MaterialTheme.colorScheme.onBackground, onSurface, onPrimary

TYPOGRAPHY:
Headers (Racing Sans One):
- headlineLarge (32sp) - App title
- headlineMedium (28sp) - Screen titles
- titleLarge (22sp) - Section headers
- titleMedium (18sp) - Card titles
- titleSmall (14sp) - Small headers

Body (Noto Sans):
- bodyLarge (16sp) - Main content
- bodyMedium (14sp) - Secondary text
- bodySmall (12sp) - Captions
- labelLarge (14sp) - Button text
- labelMedium (12sp) - Form labels
*/
